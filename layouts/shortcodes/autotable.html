{{/* This organizes a CSV file to map columns and row values. Embed formatting hints into column name, where column name is X: - X-i : type is integer - X-f : type is float - X-b : display bold - X-c : center align column - X-r : right align column - X-a : default sort ascending - X-d : default sort descending - x-nw : force values to not wrap in cell Pass multiple options accordingly: - age-n-c-b X-link for a URL field that will create a link of the text in column named X. X-sort will be used to sort column named X. Remember that if you rename columns, any corresponding X-link and X-sort columns will also need to be renamed to refer back to them properly. Ex.: name-b --> name-b-nw name-b-link --> name-b-nw-link */}} {{- $filebasename := .Get 0 -}} {{- $id := .Get 1 -}} {{- $cols := .Get 2 | lower -}} {{- $citation := .Get 3 -}} {{- $wherekey := .Get 4 | lower | default "" -}} {{- $whereop := .Get 5 | lower | default ""}} {{- $wherematch := .Get 6 | default "" -}} {{- $datefmt := .Get 7 | default "02 Jan 2006" -}} {{- $boldrownums := split ( .Get 8 ) "," -}} {{- $inner := .Inner -}} {{/* set aside this value before context (.) changes in 'with' loop */}} {{- $fullname := print $filebasename ".csv" -}} {{- $fullcsvrelref := path.Join "data" $fullname -}} {{- $csv := resources.Get $fullcsvrelref -}} {{- $rows := $csv | transform.Unmarshal (dict "delimiter" ",") -}} {{/* Guard against empty CSV */}} {{- with $rows -}} {{- $headerRow := index $rows 0 -}} {{- $datarows := after 1 $rows -}} {{- $data := slice -}} {{- range $i, $row := $datarows -}} {{- $mappedrow := dict -}} {{- range $j, $rowlabel := $headerRow -}} {{- $rowlabel = $rowlabel | lower -}} {{- $mappedrow = merge $mappedrow (dict $rowlabel (index $row $j)) -}} {{- end -}} {{- $data = $data | append $mappedrow -}} {{- end -}} {{- with (and $wherekey $whereop $wherematch ) -}} {{- $data = where $data $wherekey $whereop $wherematch -}} {{- end -}} {{- with $data -}} {{- else -}} {{- errorf "No data to display in %s autotable in %s. Looking in %s for '%s %s %s'." $id .Page.RelPermalink $filebasename $wherekey $whereop $wherematch -}} {{- end -}} {{- $displayoptions := slice "i" "f" "b" "c" "r" "a" "d" "nw" -}} {{- $aside := "" -}} {{- if $inner -}} {{- $aside = $inner -}} {{- end -}} {{- $includecols := split $cols "," -}} {{- with $data -}} {{- end -}} {{- else -}} {{- warnf "CSV is empty: %s" $fullcsvrelref -}} {{- end -}} {{- $raw := .Inner -}} {{- $out := $raw | replaceRE "(?m)\n\\s*\\s*\n" "" | replaceRE "(?m)\r" "" | safeHTML -}} {{- $out -}}
